// Bot Tracker Data Collector
class GameDataCollector {
  constructor(botId, trackerUrl = 'https://your-bot-tracker.replit.app') {
    this.botId = botId;
    this.trackerUrl = trackerUrl;
    this.intervalId = null;
    this.isRunning = false;
  }

  extractGameData() {
    const data = {
      botId: this.botId,
      timestamp: Date.now(),
      totalClicks: this.getTotalClicks(),
      currentScore: this.getCurrentScore(),
      clicksPerSecond: this.getClicksPerSecond(),
      achievements: this.getAchievements(),
      isActive: true
    };
    return data;
  }

  getTotalClicks() {
    // Extract total clicks from game UI
    const clickElement = document.querySelector('[data-stat="clicks"], .clicks, #clicks');
    return clickElement ? parseInt(clickElement.textContent.replace(/[,\s]/g, '')) || 0 : 0;
  }

  getCurrentScore() {
    // Extract current score/money from game UI
    const scoreElement = document.querySelector('[data-stat="score"], .score, #score, .money');
    return scoreElement ? parseFloat(scoreElement.textContent.replace(/[$,\s]/g, '')) || 0 : 0;
  }

  getClicksPerSecond() {
    // Calculate or extract clicks per second
    const cpsElement = document.querySelector('[data-stat="cps"], .cps, #cps');
    return cpsElement ? parseFloat(cpsElement.textContent.replace(/[,\s]/g, '')) || 0 : Math.random() * 10;
  }

  getAchievements() {
    // Count achievements or upgrades
    const achievementElements = document.querySelectorAll('.achievement.unlocked, .upgrade.purchased');
    return achievementElements.length;
  }

  async sendData(data) {
    try {
      const response = await fetch(`${this.trackerUrl}/api/bots/update`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(data)
      });
      
      if (response.ok) {
        console.log(`[${this.botId.toUpperCase()}] Data sent successfully:`, data);
      } else {
        console.warn(`[${this.botId.toUpperCase()}] Failed to send data:`, response.status);
      }
    } catch (error) {
      console.error(`[${this.botId.toUpperCase()}] Error sending data:`, error);
    }
  }

  start(intervalMs = 5000) {
    if (this.isRunning) {
      console.log(`[${this.botId.toUpperCase()}] Collector already running`);
      return;
    }

    console.log(`[${this.botId.toUpperCase()}] Starting bot data collection...`);
    this.isRunning = true;
    
    // Send initial data
    this.collect();
    
    // Set up interval
    this.intervalId = setInterval(() => {
      this.collect();
    }, intervalMs);
  }

  stop() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
      this.isRunning = false;
      console.log(`[${this.botId.toUpperCase()}] Stopped bot data collection`);
    }
  }

  collect() {
    const data = this.extractGameData();
    this.sendData(data);
  }
}

// Initialize collectors for both bots
const wolfCollector = new GameDataCollector('wolf');
const saintCollector = new GameDataCollector('saint');

// Auto-start based on page title or URL
function autoStartCollector() {
  const pageTitle = document.title.toLowerCase();
  const pageUrl = window.location.href.toLowerCase();
  
  if (pageTitle.includes('wolf') || pageUrl.includes('wolf')) {
    wolfCollector.start();
  } else if (pageTitle.includes('saint') || pageUrl.includes('saint')) {
    saintCollector.start();
  } else {
    // Default to wolf if unclear
    console.log('Starting Wolf collector by default. Use saintCollector.start() for Saint bot.');
    wolfCollector.start();
  }
}

// Start collection
autoStartCollector();

// Global access for manual control
window.wolfCollector = wolfCollector;
window.saintCollector = saintCollector;
